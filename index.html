def generacion_codigo(n):
    keyList = listaLetra = ['A', 'B', 'C', 'D', 'E']

    eleccion_numero = random.randint(0,n)
    eleccion_numero = "{0:0=4d}".format(eleccion_numero)
    codigo = eleccion_letra + "-"+ eleccion_numero
    return codigo
